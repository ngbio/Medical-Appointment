@startuml
!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

title Medical Appointment System - Client Side

' ====== Actors ======
Person(patient, "Patient", "Books and manages medical appointments")
Person(doctor, "Doctor", "Views and manages schedules")
Person(admin, "Administrator", "Manages users and system configuration")
Person(receptionist, "Receptionist", "Assists patients and manages appointment bookings")

' ====== System Boundary ======
System_Boundary(system, "Medical Appointment System") {

    Container(web, "Web Client","Web Application (React, Bootstrap 5)","Runs in the browser, provides user interface and communicates with the Backend via REST APIs")

    Container(backend, "Backend Application","Provides REST APIs, handles authentication, authorization, and business logic processing")

}

' ====== Relationships ======

Rel(patient, web, "Book appointments, manage profile", "HTTPS")

Rel(doctor, web, "View schedules and manage appointments", "HTTPS")

Rel(admin, web, "Manage users, schedules, and system settings", "HTTPS")

Rel(receptionist, web, "Create and manage appointments for patients", "HTTPS")

Rel(web, backend, "Handle requests sent from the frontend", "HTTPS/JSON")


@enduml
